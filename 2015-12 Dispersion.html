for i in range(24):
    df['mkt_%d'%i] = df.mkt.shift(-i)
    rez_std[i] = (sm.ols('mkt_%i ~ mkt + disp_std' % i,data=df.loc['1950':])
                  .fit(cov_type='HAC', cov_kwds={'maxlags': 12}, use_t=True).summary2())
    rez_ac[i] = (sm.ols('mkt_%i ~ mkt + disp_ac' % i,data=df.loc['1950':])
